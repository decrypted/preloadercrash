#Run in a docker
 `make run`

#Enable dumps (optional)
 `sudo sysctl -w kernel.core_pattern=core.%u.%p.%t`
 
#Get trace
- https://bugs.php.net/bugs-generating-backtrace.php
- `make shell` and run `gdb /usr/local/bin/php core.*` then get the bt
